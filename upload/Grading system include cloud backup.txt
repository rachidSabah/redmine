        function initializeCloudIntegrations() {
            const googleConfig = cloudConfig.google || {};
            if (googleConfig.clientId && googleConfig.apiKey) {
                document.getElementById('destGoogleDrive').disabled = false;
                document.getElementById('gdrive-connect-btn').disabled = false;
                document.getElementById('gdrive-status').textContent = 'Prêt';
                document.getElementById('gdrive-status').className = 'cloud-status status-configured';
                gapi.load('client', initializeGapiClient);
                gisInited = google.accounts.oauth2.initTokenClient({
                    client_id: googleConfig.clientId,
                    scope: googleConfig.scopes,
                    callback: '', // A blank callback avoids a redirect.
                });
            } else {
                console.warn('Clés Google Drive manquantes. La fonctionnalité est désactivée.');
            }
        }
        
        async function initializeGapiClient() {
            const googleConfig = cloudConfig.google || {};
            await gapi.client.init({
                apiKey: googleConfig.apiKey,
                discoveryDocs: ['https://www.googleapis.com/discovery/v1/apis/drive/v3/rest'],
            });
            gapiInited = true;
            console.log("Client GAPI initialisé pour Google Drive.");
        }

        function handleGoogleAuthClick() {
            const googleConfig = cloudConfig.google || {};
            if (!googleConfig.clientId) {
                showToast('Client ID Google manquant. Veuillez le configurer.', 'danger');
                return;
            }
            tokenClient = google.accounts.oauth2.initTokenClient({
                client_id: googleConfig.clientId,
                scope: googleConfig.scopes,
                callback: (tokenResponse) => {
                    if (tokenResponse.error) {
                         console.error('Erreur d\'authentification Google:', tokenResponse.error);
                         showToast(`Erreur d'authentification: ${tokenResponse.error_description}`, 'danger');
                         return;
                    }
                    console.log("Authentification Google réussie.");
                    showToast('Connecté à Google Drive avec succès!', 'success');
                    document.getElementById('gdrive-status').textContent = 'Connecté';
                    document.getElementById('gdrive-status').className = 'cloud-status status-connected';
                    document.getElementById('gdrive-connect-btn').textContent = 'Changer de compte';
                },
            });

             if (gapi.client.getToken() === null) {
                tokenClient.requestAccessToken({prompt: 'consent'});
            } else {
                tokenClient.requestAccessToken({prompt: ''});
            }
        }
        
        async function uploadToGoogleDrive(fileContent, fileName) {
            if (!gapiInited) {
                 showToast('Le client Google API n\'est pas prêt.', 'danger');
                 console.error("Tentative d'upload avant l'initialisation de GAPI.");
                 return;
            }
            const token = gapi.client.getToken();
            if (!token) {
                showToast('Veuillez vous connecter à Google Drive d\'abord.', 'warning');
                handleGoogleAuthClick(); // Prompt for login
                return;
            }

            showToast(`Début de l'upload vers Google Drive : ${fileName}`, 'info');
            console.log(`Début de l'upload de ${fileName} vers Google Drive.`);

            const metadata = {
                name: fileName,
                mimeType: 'application/json',
            };

            const form = new FormData();
            form.append('metadata', new Blob([JSON.stringify(metadata)], { type: 'application/json' }));
            form.append('file', new Blob([fileContent], {type: 'application/json'}));

            try {
                const response = await fetch('https://www.googleapis.com/upload/drive/v3/files?uploadType=multipart', {
                    method: 'POST',
                    headers: new Headers({ 'Authorization': `Bearer ${token.access_token}` }),
                    body: form,
                });
                
                const result = await response.json();
                
                if (result.error) {
                    console.error('Erreur d\'upload Google Drive:', result.error);
                    showToast(`Échec de l'upload : ${result.error.message}`, 'danger');
                } else {
                    console.log('Fichier uploadé avec succès :', result);
                    showToast('Fichier sauvegardé sur Google Drive avec succès!', 'success');
                     // Log history
                    const history = JSON.parse(localStorage.getItem('backupHistory') || '[]');
                    history.unshift({ type: 'Google Drive', date: new Date().toISOString(), file: fileName });
                    localStorage.setItem('backupHistory', JSON.stringify(history.slice(0, 10)));
                }

            } catch(error) {
                console.error('Exception lors de l\'upload sur Google Drive :', error);
                showToast(`Erreur réseau lors de l'upload: ${error.message}`, 'danger');
            }
        }

        function restoreFromLocal(event) {
            const file = event.target.files[0];
            if (!file) return;
            const reader = new FileReader();
            reader.onload = function(e) {
                try {
                    const data = JSON.parse(e.target.result);
                    if (data.students) {
                        localStorage.setItem('savedStudents', JSON.stringify(data.students));
                    }
                     if (data.settings) {
                        localStorage.setItem('schoolLogoLeft', data.settings.logoLeft || '');
                        localStorage.setItem('schoolLogoRight', data.settings.logoRight || '');
                        localStorage.setItem('docxFooterText', data.settings.footer || '');
                    }
                    loadSettings();
                    populateAcademicYearFilter();
                    loadSavedStudents();
                    showToast('Données restaurées avec succès!', 'success');
                } catch (error) {
                    showToast('Erreur lors de la restauration: ' + error.message, 'danger');
                }
            };
            reader.readAsText(file);
            event.target.value = '';
        }

        function importStudentsFromCSV(event) {
            const file = event.target.files[0];
            if (!file) return;

            const reader = new FileReader();
            reader.onload = function(e) {
                try {
                    const csv = e.target.result;
                    const lines = csv.split('\n').map(l => l.trim()).filter(Boolean);
                    if (lines.length < 2) throw new Error("CSV file is empty or has no data rows.");
                    
                    const header = lines[0].split(',').map(h => h.trim().replace(/"/g, ''));
                    const requiredHeaders = ["Nom et Prénom", "N° Inscription", "Promotion", "Date"];
                    if (!requiredHeaders.every(h => header.includes(h))) {
                        throw new Error(`CSV must contain the following headers: ${requiredHeaders.join(', ')}`);
                    }

                    const savedStudents = JSON.parse(localStorage.getItem('savedStudents')) || {};
                    let importedCount = 0;

                    for (let i = 1; i < lines.length; i++) {
                        const values = lines[i].split(',').map(v => v.trim().replace(/"/g, ''));
                        const studentObj = {};
                        header.forEach((h, index) => {
                            studentObj[h] = values[index] ? values[index].trim() : '';
                        });

                        const studentName = studentObj["Nom et Prénom"];
                        if (studentName) {
                            savedStudents[studentName] = {
                                name: studentName,
                                registration: studentObj["N° Inscription"] || '',
                                promotion: studentObj["Promotion"] || '',
                                date: studentObj["Date"] || '',
                                soutenance: 0,
                                modules1: [],
                                modules2: [],
                                year1: null,
                                year2: null,
                            };
                            importedCount++;
                        }
                    }
                    
                    localStorage.setItem('savedStudents', JSON.stringify(savedStudents));
                    populateAcademicYearFilter();
                    loadSavedStudents();
                    alert(`${importedCount} students imported successfully!`);

                } catch (error) {
                    alert('Error importing CSV: ' + error.message);
                } finally {
                    event.target.value = ''; // Reset file input
                }
            };
            reader.readAsText(file);
        }

        function calculateYearResults(modules, year, year1Results = null, soutenance = 0) {
             if (!modules || modules.length === 0) return { mcc: 0, mefcft: 0, mefcfp: 0, average: 0, finalGrade: 0 };
            const mcc = modules.reduce((sum, m) => sum + (parseFloat(m.cc) || 0), 0) / modules.length;
            const mefcft = modules.reduce((sum, m) => sum + (parseFloat(m.efcfT) || 0), 0) / modules.length;
            const mefcfp = modules.reduce((sum, m) => sum + (parseFloat(m.efcfP) || 0), 0) / modules.length;
            const average = (mcc * 3 + mefcft * 2 + mefcfp * 3) / 8;
            let finalGrade = 0;
            if (year1Results) {
                const year2mcc = (year === 2) ? mcc : 0;
                const year2mefcft = (year === 2) ? mefcft : 0;
                const year2mefcfp = (year === 2) ? mefcfp : 0;
                const hasYear2Modules = modules2 && modules2.length > 0;
                const combinedCC = hasYear2Modules ? (year1Results.mcc + year2mcc) / 2 : year1Results.mcc;
                const combinedEFCFT = hasYear2Modules ? (year1Results.mefcft + year2mefcft) / 2 : year1Results.mefcft;
                const combinedEFCFP = hasYear2Modules ? (year1Results.mefcfp + year2mefcfp) / 2 : year1Results.mefcfp;
                finalGrade = (combinedCC * 3 + combinedEFCFT * 2 + combinedEFCFP * 3 + parseFloat(soutenance) * 2) / 10;
            }
            return { mcc, mefcft, mefcfp, average, finalGrade };
        }

        function convertHtmlToDocxObjects(htmlString) {
            const parser = new DOMParser();
            const doc = parser.parseFromString(`<div>${htmlString}</div>`, 'text/html');
            const paragraphs = [];

            function processNode(node) {
                let runs = [];
                for (const childNode of node.childNodes) {
                    if (childNode.nodeType === Node.TEXT_NODE) {
                         runs.push(new docx.TextRun(childNode.textContent));
                    } else if (childNode.nodeType === Node.ELEMENT_NODE) {
                       let style = {};
                        if (childNode.tagName === 'B' || childNode.tagName === 'STRONG') {
                            style.bold = true;
                        }
                        if (childNode.tagName === 'I' || childNode.tagName === 'EM') {
                            style.italics = true;
                        }
                        if (childNode.tagName === 'U') {
                           style.underline = {};
                        }
                        
                        const childRuns = processNode(childNode);
                        childRuns.forEach(run => {
                           if(run.options){
                               Object.assign(run.options, style);
                           } else {
                               run.options = style;
                           }
                        });
                        runs.push(...childRuns);
                    }
                }
                return runs;
            }

            for(const p of doc.body.children){
                 paragraphs.push(new docx.Paragraph({
                    children: processNode(p),
                    alignment: docx.AlignmentType.CENTER // Default to center for footer
                }));
            }
             if(doc.body.children.length === 0 && doc.body.textContent){
                 paragraphs.push(new docx.Paragraph({
                    children: processNode(doc.body),
                    alignment: docx.AlignmentType.CENTER 
                }));
            }

            return paragraphs;
        }
        
        async function createDocxHeader() {
            let headerChildren = [];
            const logoLeftBase64 = localStorage.getItem('schoolLogoLeft');
            const logoRightBase64 = localStorage.getItem('schoolLogoRight');
            let leftLogoImage, rightLogoImage;

            if (logoLeftBase64) {
                const response = await fetch(logoLeftBase64);
                const imageBuffer = await response.arrayBuffer();
                leftLogoImage = new docx.ImageRun({ data: imageBuffer, transformation: { width: 190, height: 88 } });
            }

            if (logoRightBase64) {
                const response = await fetch(logoRightBase64);
                const imageBuffer = await response.arrayBuffer();
                rightLogoImage = new docx.ImageRun({ data: imageBuffer, transformation: { width: 90, height: 88 } });
            }

            const logoTable = new docx.Table({
                width: { size: 100, type: docx.WidthType.PERCENTAGE },
                borders: docx.TableBorders.NONE,
                rows: [new docx.TableRow({
                    children: [
                        new docx.TableCell({
                            children: [leftLogoImage ? new docx.Paragraph({ children: [leftLogoImage] }) : new docx.Paragraph('')],
                            width: { size: 50, type: docx.WidthType.PERCENTAGE }
                        }),
                        new docx.TableCell({
                            children: [rightLogoImage ? new docx.Paragraph({ children: [rightLogoImage], alignment: docx.AlignmentType.RIGHT }) : new docx.Paragraph('')],
                            width: { size: 50, type: docx.WidthType.PERCENTAGE }
                        })
                    ]
                })]
            });
            headerChildren.push(logoTable);
            return headerChildren;
        }

        function makeHeaderCell(text, options = {}) {
            return new docx.TableCell({
                children: [new docx.Paragraph({ 
                    children: [new docx.TextRun({ text: String(text), bold: true })],
                    alignment: docx.AlignmentType.CENTER 
                })],
                shading: { fill: "D3D3D3", val: docx.ShadingType.CLEAR, color: "auto" },
                verticalAlign: docx.VerticalAlign.CENTER,
                ...options
            });
        }

        function importData(event) {
            const file = event.target.files[0];
            if (!file) return;
            const reader = new FileReader();
            reader.onload = function(e) {
                try {
                    JSON.parse(e.target.result); 
                    localStorage.setItem('savedStudents', e.target.result);
                    populateAcademicYearFilter();
                    loadSavedStudents();
                    alert('Data imported successfully!');
                } catch (error) {
                    alert('Error importing data: ' + error.message);
                }
            };
            reader.readAsText(file);
            event.target.value = '';

        function splitNomPrenom(fullName){ if(!fullName) return { nom:'', prenom:'' }; const parts=fullName.trim().split(/\s+/); if(parts.length===1) return { nom:parts[0], prenom:'' }; const prenom=parts.pop(); return { nom:parts.join(' '), prenom }; }
        function safe2(n){ return (Number(n)||0).toFixed(2); }
        function calcYear1FromStudent(s) { let y1 = s.year1; if (!y1 || typeof y1.mcc !== 'number') { const mods = s.modules1 || []; if (mods.length) { const mcc = mods.reduce((a, m) => a + (Number(m.cc) || 0), 0) / mods.length; const mefcft = mods.reduce((a, m) => a + (Number(m.efcfT) || 0), 0) / mods.length; const mefcfp = mods.reduce((a, m) => a + (Number(m.efcfP) || 0), 0) / mods.length; const average = (mcc * 3 + mefcft * 2 + mefcfp * 3) / 8; y1 = { mcc, mefcft, mefcfp, average }; } else { y1 = { mcc: 0, mefcft: 0, mefcfp: 0, average: 0 }; } } const wcc = y1.mcc * 0.30; const wet = y1.mefcft * 0.20; const wep = y1.mefcfp * 0.30; const moyenne = (wcc + wet + wep) / 0.80; return { ...y1, wcc, wet, wep, moyenne }; }
        function calcYear2FromStudent(s) { const y1 = calcYear1FromStudent(s); let y2 = s.year2; if (!y2 || typeof y2.mcc !== 'number') { const mods = s.modules2 || []; if (mods.length) { const mcc = mods.reduce((a, m) => a + (Number(m.cc) || 0), 0) / mods.length; const mefcft = mods.reduce((a, m) => a + (Number(m.efcfT) || 0), 0) / mods.length; const mefcfp = mods.reduce((a, m) => a + (Number(m.efcfP) || 0), 0) / mods.length; const average = (mcc * 3 + mefcft * 2 + mefcfp * 3) / 8; y2 = { mcc, mefcft, mefcfp, average }; } else { y2 = { mcc: 0, mefcft: 0, mefcfp: 0, average: 0 }; } } const hasYear2Modules = s.modules2 && s.modules2.length > 0; const combinedCC = hasYear2Modules ? (y1.mcc + y2.mcc) / 2 : y1.mcc; const combinedEFCFT = hasYear2Modules ? (y1.mefcft + y2.mefcft) / 2 : y1.mefcft; const combinedEFCFP = hasYear2Modules ? (y1.mefcfp + y2.mefcfp) / 2 : y1.mefcfp; const sout = Number(s.soutenance || 0); const finalGrade = (combinedCC * 3 + combinedEFCFT * 2 + combinedEFCFP * 3 + sout * 2) / 10; const wcc = combinedCC * 0.30; const wet = combinedEFCFT * 0.20; const wep = combinedEFCFP * 0.30; const wsout = sout * 0.20; return { y1, y2, combinedCC, combinedEFCFT, combinedEFCFP, finalGrade, wcc, wet, wep, wsout }; }
