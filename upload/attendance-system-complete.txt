        function generateSampleAttendance() {
            const today = new Date();
            
            // Generate attendance for the last 7 days
            for (let i = 0; i < 7; i++) {
                const date = new Date(today);
                date.setDate(date.getDate() - i);
                const dateStr = date.toISOString().split('T')[0];
                
                if (!attendanceRecords[dateStr]) {
                    attendanceRecords[dateStr] = {};
                }
                
                students.forEach(student => {
                    const random = Math.random();
                    let status = 'present';
                    let time = '08:' + Math.floor(Math.random() * 30).toString().padStart(2, '0');
                    
                    if (random < 0.05) {
                        status = 'absent';
                        time = '';
                    } else if (random < 0.1) {
                        status = 'late';
                        time = '08:' + (30 + Math.floor(Math.random() * 30)).toString().padStart(2, '0');
                    }
                    
                    attendanceRecords[dateStr][student.id] = {
                        status: status,
                        time: time,
                        notes: status === 'late' ? 'Traffic delay' : '',
                        recordedAt: new Date().toISOString()
                    };
                });
            }
            
            saveToLocalStorage();
        }

        function sendWhatsAppMessage(studentId, messageType) {
            if (!whatsappEnabled) return;

            const student = students.find(s => s.id === studentId);
            if (!student || !student.parentPhone) return;

            const settings = JSON.parse(localStorage.getItem('whatsappSettings')) || {};
            let template = messageType === 'absent' ? settings.absentTemplate : settings.lateTemplate;
            
            if (!template) {
                template = `Hello, your child ${student.name} was marked ${messageType} today.`;
            }

            // Replace template variables
            const message = template
                .replace('{student_name}', student.name)
                .replace('{student_id}', student.id)
                .replace('{parent_name}', 'Parent')
                .replace('{date}', currentDate)
                .replace('{school_name}', 'AttendanceHUB Academy');

            // Create WhatsApp URL
            const whatsappUrl = `https://wa.me/${student.parentPhone.replace(/[^0-9]/g, '')}?text=${encodeURIComponent(message)}`;
            
            // Open WhatsApp
            window.open(whatsappUrl, '_blank');
            
            // Log message
            const messageLog = {
                date: new Date().toISOString(),
                student: student.name,
                parentPhone: student.parentPhone,
                type: messageType,
                status: 'sent'
            };
            
            whatsappMessages.push(messageLog);
            saveToLocalStorage();
            loadMessageHistory();
            
            showNotification(`WhatsApp message sent to ${student.name}'s parent`, 'success');
        }

        function sendCustomWhatsAppMessage() {
            const recipients = document.getElementById('message-recipients').value;
            const message = document.getElementById('custom-message').value;
            
            if (!message.trim()) {
                showNotification('Please enter a message', 'error');
                return;
            }
            
            let targetStudents = [];
            
            if (recipients === 'all') {
                targetStudents = students;
            } else if (recipients === 'class') {
                const selectedClass = document.getElementById('message-class').value;
                targetStudents = students.filter(s => s.class === selectedClass);
            } else if (recipients === 'absent') {
                const todayRecords = attendanceRecords[currentDate] || {};
                targetStudents = students.filter(s => 
                    todayRecords[s.id] && todayRecords[s.id].status === 'absent'
                );
            }
            
            targetStudents.forEach(student => {
                if (student.parentPhone) {
                    const whatsappUrl = `https://wa.me/${student.parentPhone.replace(/[^0-9]/g, '')}?text=${encodeURIComponent(message)}`;
                    setTimeout(() => {
                        window.open(whatsappUrl, '_blank');
                    }, 1000);
                }
            });
            
            showNotification(`Message will be sent to ${targetStudents.length} parents`, 'success');
        }

        function updateAttendanceStatus(studentId, status) {
            const previousRecord = attendanceRecords[currentDate] && attendanceRecords[currentDate][studentId];
            const previousStatus = previousRecord ? previousRecord.status : null;
            
            updateAttendanceRecord(studentId, { status: status });
            
            // Send WhatsApp message if status changed to absent and auto-send is enabled
            if (status === 'absent' && previousStatus !== 'absent') {
                const autoSend = JSON.parse(localStorage.getItem('whatsappSettings'))?.autoSendAbsent;
                if (whatsappEnabled && autoSend) {
                    sendWhatsAppMessage(studentId, 'absent');
                }
            }
        }

        function updateAttendanceRecord(studentId, updates) {
            const selectedDate = document.getElementById('attendance-date').value;
            
            if (!attendanceRecords[selectedDate]) {
                attendanceRecords[selectedDate] = {};
            }
            
            if (!attendanceRecords[selectedDate][studentId]) {
                attendanceRecords[selectedDate][studentId] = {};
            }
            
            Object.assign(attendanceRecords[selectedDate][studentId], updates);
            attendanceRecords[selectedDate][studentId].recordedAt = new Date().toISOString();
            
            saveToLocalStorage();
            updateDashboardStats();
        }

        function handleStudentImport(input) {
            const file = input.files[0];
            if (file) {
                const reader = new FileReader();
                reader.onload = function(e) {
                    try {
                        const csv = e.target.result;
                        const lines = csv.split('\n');
                        const headers = lines[0].split(',');
                        
                        let importedCount = 0;
                        for (let i = 1; i < lines.length; i++) {
                            if (lines[i].trim()) {
                                const values = lines[i].split(',');
                                const student = {
                                    id: values[0] || generateStudentId(),
                                    name: values[1],
                                    class: values[2],
                                    email: values[3] || '',
                                    phone: values[4] || '',
                                    parentPhone: values[5] || '',
                                    dob: values[6] || '',
                                    address: values[7] || '',
                                    status: values[8] || 'active'
                                };
                                
                                if (student.name && student.class) {
                                    // Check if student already exists
                                    if (!students.some(s => s.id === student.id)) {
                                        students.push(student);
                                        importedCount++;
                                    }
                                }
                            }
                        }
                        
                        saveToLocalStorage();
                        loadStudentsTable();
                        updateDashboardStats();
                        showNotification(`${importedCount} students imported successfully`, 'success');
                    } catch (error) {
                        showNotification('Error importing file. Please check the format.', 'error');
                    }
                };
                reader.readAsText(file);
            }
        }

            reader.onload = function(e) {
                try {
                    const backupData = JSON.parse(e.target.result);
                    
                    // Validate backup data
                    if (!backupData.students || !backupData.teachers || !backupData.classes) {
                        throw new Error('Invalid backup file format');
                    }
                    
                    if (confirm('This will replace all current data. Continue?')) {
                        students = backupData.students || [];
                        teachers = backupData.teachers || [];
                        classes = backupData.classes || [];
                        attendanceRecords = backupData.attendanceRecords || {};
                        whatsappMessages = backupData.whatsappMessages || [];
                        
                        saveToLocalStorage();
                        loadAllTables();
                        populateAllDropdowns();
                        updateDashboardStats();
                        
                        showNotification('Backup restored successfully', 'success');
                        addRecentActivity('System backup restored');
                    }
                    
                } catch (error) {
                    showNotification('Invalid backup file', 'error');
                }
            };
            reader.readAsText(file);
        }

        function toggleAutoBackup() {
            const enabled = document.getElementById('auto-backup-enabled').checked;
            showNotification(`Auto backup ${enabled ? 'enabled' : 'disabled'}`, 'info');
        }

        function saveBackupSettings() {
            showNotification('Backup settings saved', 'success');
        }

        // Settings Functions
        function saveSettings() {
            showNotification('Settings saved successfully', 'success');
        }

        // Utility Functions
        function populateAllDropdowns() {
            populateClassDropdowns();
            populateClassTeachers();
        }

        function populateClassDropdowns() {
            const dropdowns = ['student-class', 'attendance-class', 'report-class', 'message-class'];
            
            dropdowns.forEach(dropdownId => {
                const dropdown = document.getElementById(dropdownId);
                if (dropdown) {
                    // Save current value
                    const currentValue = dropdown.value;
                    
                    // Clear options except first
                    while (dropdown.children.length > 1) {
                        dropdown.removeChild(dropdown.lastChild);
                    }
                    
                    // Add class options
                    classes.forEach(classItem => {
                        const option = document.createElement('option');
                        option.value = classItem.name;
                        option.textContent = classItem.name;
                        dropdown.appendChild(option);
                    });
                    
                    // Restore value
                    dropdown.value = currentValue;
                }
            });
        }

        function populateClassTeachers() {
            const dropdown = document.getElementById('class-teacher');
            if (dropdown) {
                dropdown.innerHTML = '<option value="">Select Teacher</option>';
                
                teachers.forEach(teacher => {
                    const option = document.createElement('option');
                    option.value = teacher.name;
                    option.textContent = teacher.name;
                    dropdown.appendChild(option);
                });
            }
        }

        function loadAllTables() {
            loadStudentsTable();
            loadTeachersTable();
            loadClassesTable();
            loadAttendanceTable();
        }

        // Recent Activity Functions
        function loadRecentActivity() {
            const activityContainer = document.getElementById('recent-activity');
            const recentActivities = JSON.parse(localStorage.getItem('recentActivities')) || [];
            
            if (recentActivities.length === 0) {
                activityContainer.innerHTML = `
                    <p style="text-align: center; color: var(--text-light); padding: 40px;">
                        <i class="fas fa-clock" style="font-size: 48px; margin-bottom: 15px; display: block;"></i>
                        No recent activity
                    </p>
                `;
                return;
            }
            
            activityContainer.innerHTML = recentActivities.slice(0, 5).map(activity => `
                <div style="padding: 15px 0; border-bottom: 1px solid var(--border);">
                    <div style="font-weight: 500; color: var(--dark);">${activity.text}</div>
                    <div style="font-size: 12px; color: var(--text-light); margin-top: 5px;">
                        <i class="fas fa-clock"></i> ${activity.timestamp}
                    </div>
                </div>
            `).join('');
        }

        function addRecentActivity(activity) {
            let recentActivities = JSON.parse(localStorage.getItem('recentActivities')) || [];
            
            const newActivity = {
                text: activity,
                timestamp: new Date().toLocaleString()
            };
            
            recentActivities.unshift(newActivity);
            recentActivities = recentActivities.slice(0, 10); // Keep only last 10 activities
            
            localStorage.setItem('recentActivities', JSON.stringify(recentActivities));
            loadRecentActivity();
        }

        function refreshActivity() {
            loadRecentActivity();
            showNotification('Activity refreshed', 'info');
        }

        // Notification System
        function showNotification(message, type) {
            const notification = document.getElementById('notification');
            notification.textContent = message;
            notification.className = `notification ${type}`;
            notification.style.display = 'block';
            
            setTimeout(() => {
                notification.style.display = 'none';
            }, 3000);
        }

        // Local Storage Functions
        function saveToLocalStorage() {
            localStorage.setItem('students', JSON.stringify(students));
            localStorage.setItem('teachers', JSON.stringify(teachers));
            localStorage.setItem('classes', JSON.stringify(classes));
            localStorage.setItem('attendanceRecords', JSON.stringify(attendanceRecords));
            localStorage.setItem('whatsappMessages', JSON.stringify(whatsappMessages));
        }

        function loadFromLocalStorage() {
            const savedStudents = localStorage.getItem('students');
            const savedTeachers = localStorage.getItem('teachers');
            const savedClasses = localStorage.getItem('classes');
            const savedAttendance = localStorage.getItem('attendanceRecords');
            const savedMessages = localStorage.getItem('whatsappMessages');
            
            if (savedStudents) {
                students = JSON.parse(savedStudents);
            }
            
            if (savedTeachers) {
                teachers = JSON.parse(savedTeachers);
            }
            
            if (savedClasses) {
                classes = JSON.parse(savedClasses);
            }
            
            if (savedAttendance) {
                attendanceRecords = JSON.parse(savedAttendance);
            }
            
            if (savedMessages) {
                whatsappMessages = JSON.parse(savedMessages);
            }
        }

        // Auto-save functionality
        setInterval(() => {
            saveToLocalStorage();
        }, 30000); // Save every 30 seconds

                }
